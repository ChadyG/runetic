;Chad Godsey
;iPhone game prototype

WIDTH = 150
HEIGHT = 150
PAUSE = 0

%%

INIT = {0,0,0}
EMPTY = {0,0,0}
FLESH = {255,0,0}
VIRUS1 = {0,250,0}
VIRUS2 = {0,150,0}
VIRUS3 = {0,50,0}
DECAY = {0,0,255}

P1 = {255,0,255}
P2 = {0,255,255}

%%

DISCRETE PLAYER1
DISCRETE FIELD
DISCRETE PLAYER2

%%

;Initialize
PLAYER1:INIT -> EMPTY*0.1
PLAYER1:INIT -> VIRUS1*0.1
PLAYER1:INIT -> VIRUS2*0.1
PLAYER1:INIT -> VIRUS3*0.1

FIELD:INIT -> FLESH

PLAYER2:INIT -> EMPTY*0.1
PLAYER2:INIT -> VIRUS1*0.1
PLAYER2:INIT -> VIRUS2*0.1
PLAYER2:INIT -> VIRUS3*0.1


;Player one field
PLAYER1:VIRUS1 + 3VIRUS2+ -> VIRUS2
PLAYER1:VIRUS2 + 3VIRUS3+ -> VIRUS3
PLAYER1:VIRUS3 + 3DECAY+ -> DECAY
PLAYER1:VIRUS3 + 3EMPTY+ -> DECAY
PLAYER1:DECAY + 3EMPTY+ -> EMPTY
PLAYER1:EMPTY + 3VIRUS1+ -> VIRUS1

;Empty decay
PLAYER1:VIRUS2 + 3EMPTY+ -> VIRUS3
PLAYER1:VIRUS3 + 3EMPTY+ -> DECAY

;Suffocation
PLAYER1:VIRUS1 + 6VIRUS1+ -> VIRUS2
PLAYER1:VIRUS2 + 6VIRUS2+ -> VIRUS3
PLAYER1:VIRUS3 + 6VIRUS3+ -> DECAY
PLAYER1:DECAY + 6DECAY+ -> EMPTY


;Player two field
PLAYER2:VIRUS1 + 3VIRUS2+ -> VIRUS2
PLAYER2:VIRUS2 + 3VIRUS3+ -> VIRUS3
PLAYER2:VIRUS3 + 3DECAY+ -> DECAY
PLAYER2:VIRUS3 + 3EMPTY+ -> DECAY
PLAYER2:DECAY + 3EMPTY+ -> EMPTY
PLAYER2:EMPTY + 3VIRUS1+ -> VIRUS1

;Empty decay
PLAYER2:VIRUS2 + 3EMPTY+ -> VIRUS3
PLAYER2:VIRUS3 + 3EMPTY+ -> DECAY

;Suffocation
PLAYER2:VIRUS1 + 6VIRUS1+ -> VIRUS2
PLAYER2:VIRUS2 + 6VIRUS2+ -> VIRUS3
PLAYER2:VIRUS3 + 6VIRUS3+ -> DECAY
PLAYER2:DECAY + 6DECAY+ -> EMPTY


;Playing field
FIELD:FLESH + PLAYER1:DECAY -> P1

FIELD:FLESH + PLAYER2:DECAY -> P2

;Round it out
FIELD:P1 + 5P2+ -> P2
FIELD:P2 + 5P1+ -> P1
